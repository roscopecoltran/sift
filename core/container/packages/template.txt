package packages

import (
	"fmt"
	"github.com/roscopecoltran/sniperkit-sift/core/sift"
)

type PACKAGE struct{}

func (package PACKAGE) URL(version string) string {
	return fmt.Sprintf("", version)
}

func (package PACKAGE) Build(config sift.Config) error {
	configure := sift.ConfigureCmd{
		Prefix: config.Prefix,
		Args: []string{
		},
	}.Cmd()
	if err := configure.Run(); err != nil {
		return err
	}
	make := sift.MakeCmd{Jobs: config.NumCores}.Cmd()
	return make.Run()
}

func (package PACKAGE) Install(config sift.Config) error {
	make := sift.MakeCmd{
		Args: []string{
			"install",
			"prefix=" + config.InstallDir("libevent"),
		},
	}.Cmd()
	return make.Run()
}

func (package PACKAGE) Dependencies() []string {
	return []string{}
}
